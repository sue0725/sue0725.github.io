    import java.util.Scanner;

    public class Main{
        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
        
            int C = sc.nextInt();
        
            for(int i = 1; i <= C; i++) {
                for(int j = 1; j <= i; j++) {
                    int A = sc.nextInt();
                    int B = sc.nextInt();
                    System.out.println("Case #" + j + ":" + A+B);
                }
            }
        }
    }
